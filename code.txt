//This code will cause an LED connected to GPIO D7 to blink the word nigel in morse code.


//GPIO pin
int led = D7;

//time values
int dit = 100;
int da = 300;
int intragap = 100;
int charGap = 300;

void setup() {

  pinMode(led, OUTPUT); // set pin to output
}

//function to have the LED perform a Dit likedisplay
void DoDit(){
    digitalWrite(led, HIGH);
    delay(dit);
    digitalWrite(led, LOW);
    delay(intragap);
}

//function to have the LED perform a Da like display
void DoDa(){
    digitalWrite(led, HIGH);
    delay(da);
    digitalWrite(led, LOW);
    delay(intragap);
}

void loop() {
  
  //N
  DoDa(); DoDit();
  //Character Space
  delay(charGap);
  
  //I
  DoDit(); DoDit();
  //Character Space
  delay(charGap);
  
  //G
  DoDa(); DoDa(); DoDit();
  //Character Space
  delay(charGap);
  
  //E
  DoDit();
  //Character Space
  delay(charGap);
  
  //L
  DoDit(); DoDa(); DoDit(); DoDit(); 

  //wait five before repeat name
  delay(5000);
}
